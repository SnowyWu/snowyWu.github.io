# 双向数据绑定

在count++例子中,随着按钮点击count的值也会发生变化,这就是Vue3中的新函数ref和react

# ref函数

//例如,可以通过ts指定泛型,也可以使用vue的类型自动推导
const name1 = ref<string>("")

const name2 = ref("")

//返回一个具有代理的对象
export declare interface Ref<T = any> {
    value: T;
    
    [RefSymbol]: true;
}
export declare function ref<T extends object>(value: T): [T] extends [Ref] ? T : Ref<UnwrapRef<T>>;

export declare function ref<T>(value: T): Ref<UnwrapRef<T>>;

export declare function ref<T = any>(): Ref<T | undefined>;

ref函数返回的对象在script中用加上.value引用即 name1.value = "张三"
